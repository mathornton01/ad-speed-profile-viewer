% ---------- THE QUESTION ----------
\begin{frame}{Why position-resolved speed?}
\begin{columns}[T]
\begin{column}{0.5\textwidth}
  \vspace{0.4em}
  \color{KWhite}\small
  RNA Polymerase II does not elongate at a constant velocity.
  Over a single gene body it \creambf{pauses}, \creambf{accelerates}, and
  \creambf{decelerates} under competing molecular forces:
  \begin{itemize}
    \item nucleosome positioning and histone marks,
    \item DNA sequence context and GC skew,
    \item cotranscriptional splicing signals,
    \item DNA-damage and R-loop formation,
    \item exonic / intronic boundaries.
  \end{itemize}
  \vspace{0.2em}
  \color{KCream}\itshape
  A single gene-wide rate erases the kinetic landscape that makes transcription
  biologically informative.
\end{column}

% ---------- THE 1/x PRINCIPLE ----------
\begin{frame}{The $1/v$ principle}
\begin{columns}[T]
\begin{column}{0.52\textwidth}
  \vspace{0.3em}
  \color{KWhite}\small
  If Pol II moves through position $x$ at speed $v(x)$, the \goldbf{dwell time}
  density is proportional to $1/v(x)$.
  \medskip

  In a nascent-RNA snapshot after elapsed time $t$, the read pile-up at
  position $x$ satisfies:
  \[
     \mathbb{E}[\,\text{reads}(x,t)\,] \;\propto\; \frac{t}{v(x)} \quad \Longrightarrow\quad
     \widehat{v}(x) \;\propto\; \frac{t}{\text{reads}(x,t)}
  \]
  \medskip

  {\color{KCream}\itshape Slow regions look dark with reads; fast regions look
  sparse --- the inverse relationship is the kinetic signature.}
\end{column}
\begin{column}{0.48\textwidth}
  \centering
  \begin{tikzpicture}[scale=0.85]
    \begin{axis}[
      width=7cm,height=5cm,
      axis lines=left,
      xlabel={\color{KCream}\scriptsize position $x$ (kb)},
      ylabel={\color{KCream}\scriptsize $v(x)$ vs.\ reads},
      axis line style={KRule,line width=0.5pt},
      tick style={KRule},
      xtick={0,2,4,6,8}, ytick=\empty,

\end{columns}

% ---------- PILLAR 1: detail ----------
\begin{frame}{Positional regression: the estimator}
\begin{columns}[T]
\begin{column}{0.52\textwidth}
  \color{KWhite}\footnotesize
  For gene $g$, position $x$, and time points $t_i \in \{0,10,25,40\}$:
  \[
    r_g(x, t_i) \;=\; \alpha_g(x) + \beta_g(x)\,t_i + \varepsilon_{i}
  \]
  The local slope $\widehat\beta_g(x)$ is the \goldbf{positional
  accumulation rate} (PAR). Because dwell time is $\propto 1/v(x)$,
  \[
    \widehat v_g(x) \;\propto\; 1/\widehat \beta_g(x).
  \]
  \begin{itemize}
    \item Robust M-estimator (Huber loss).
    \item Positions smoothed with a $\pm$250\,bp window.
    \item Aggregated to gene level via algebraic diversity $Z_M$.
  \end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
  \centering
  \begin{tikzpicture}[scale=0.85]
    \begin{axis}[
      width=7.1cm,height=5.2cm,
      axis lines=left,
      xlabel={\color{KCream}\scriptsize time (min)},

      axis line style={KRule},
      tick style={KRule},

% ---------- ALGEBRAIC DIVERSITY BRIDGE ----------
\begin{frame}{From position to gene: algebraic diversity}
\begin{columns}[T]
\begin{column}{0.55\textwidth}
  \color{KWhite}\small
  Per-position slopes are noisy. To get a \goldbf{gene-level} speed we
  borrow the algebraic-diversity framework:
  \begin{itemize}
    \item Treat $\{\widehat\beta(x)\}_{x\in g}$ as elements of an abelian group.
    \item Estimator: $Z_M = \dfrac{1}{M-1}\sum_{x} \mathbf{1}[\widehat\beta(x)\ne 0]
                       \cdot f(\widehat\beta(x))$.
    \item Spectral concentration $\psi$ measures \skyit{how tightly} the
          position-signal concentrates on one dominant mode.
  \end{itemize}
  \medskip
  In practice $\psi$ on gene $g$ is:
  \[
    \psi(g) \;=\; \frac{\lambda_{1}^{2}}{\sum_j \lambda_j^{2}},
    \quad \lambda_j = \text{SVD of the position-time matrix.}
  \]
\end{column}
\begin{column}{0.45\textwidth}
  \centering
  \begin{tikzpicture}
    \node[rectangle,fill=KNavyLight,draw=KRule,rounded corners=2pt,
          inner sep=10pt,text width=6.5cm]{%

% ---------- PILLAR 2: HMM validation ----------
\begin{frame}{The DANKO wave-front HMM, reimplemented}
\begin{columns}[T]
\begin{column}{0.5\textwidth}
  \color{KWhite}\small
  \goldbf{What DANKO does.} Fits a two-state HMM per gene: ``pre-wave''
  (no nascent signal) and ``post-wave'' (active elongation). The breakpoint
  location at time $t$ yields the gene-wide rate $v_g$.\\[0.4em]
  \goldbf{What we built.}
  \begin{itemize}
    \item End-to-end Python port of \texttt{groHMM} (previously R-only).
    \item Baum--Welch + Viterbi per gene, per time point.
    \item Parallelised over genes: full cohort in $\sim$7 min on 1 GPU node.
    \item Public package bundled in repo: \texttt{kairos.grohmm}.
  \end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
  \centering
  \includegraphics[width=\linewidth]{grohmm_vs_danko_comparison.png}
  \vspace{0.2em}

  {\color{KCream}\scriptsize\itshape Python port vs.\ original R groHMM: near-identical rate recovery.}
\end{column}
\end{columns}
\end{frame}

% ---------- PILLAR 3: epigenetics ----------
\begin{frame}{Epigenetic covariates: the design}
\begin{columns}[T]
\begin{column}{0.55\textwidth}
  \color{KWhite}\small
  For each position $x$ in each gene $g$ we assemble an epigenetic feature vector
  $\mathbf{z}_g(x) \in \mathbb{R}^{K}$:
  \begin{itemize}
    \item H3K4me3, H3K27ac, H3K36me3, H3K9me3, H3K27me3
    \item DNase-seq accessibility
    \item WGBS methylation ($\beta$-value)
    \item GC content, CpG density
    \item distance to nearest exon boundary
  \end{itemize}
  \medskip
  Mixed-effects model:
  \[
    \log \widehat v_g(x) \;=\; \mathbf{z}_g(x)^\top \boldsymbol\gamma
       + u_g + \eta(x) + \varepsilon
  \]
  $u_g$ = gene random effect, $\eta(x)$ = positional spline.
\end{column}
\begin{column}{0.45\textwidth}
  \centering
  \begin{tikzpicture}[scale=0.9]
    \node[rectangle,fill=KNavyLight,draw=KRule,inner sep=8pt,text width=6.5cm]{%

      H3K36me3   & $+0.22$ \\
      H3K27ac    & $+0.14$ \\

      GC content & $-0.08$ \\

% ---------- APPENDIX: formulas reference ----------
\begin{frame}{Appendix: formulas at a glance}
\small
\color{KWhite}
\begin{align*}
\text{Positional regression:} \quad &
   r_g(x,t) = \alpha_g(x) + \beta_g(x)\,t + \varepsilon,\;\;
   \widehat v_g(x) \propto 1/\widehat\beta_g(x) \\[0.2em]
\text{Algebraic diversity:} \quad &
   Z_M \;=\; \frac{1}{M-1}\sum_{x=1}^{M} \mathbf{1}[\widehat\beta(x) \ne 0]\,f(\widehat\beta(x)) \\[0.2em]
\text{Spectral concentration:} \quad &
   \psi(g) \;=\; \lambda_1^2 \big/ \textstyle\sum_j \lambda_j^2,\;\;
   \{\lambda_j\} = \text{SVD of $R_g \in \mathbb{R}^{M\times T}$} \\[0.2em]
\text{Epigenetic model:} \quad &
   \log \widehat v_g(x) \;=\; \mathbf{z}_g(x)^\top \boldsymbol\gamma + u_g + \eta(x) + \varepsilon \\[0.2em]
\text{Validation anchor:} \quad &
   v_g^{\text{DANKO}} \;=\; \text{wave-front location at time $t$} \big/ t
\end{align*}
\vspace{0.4em}
\begin{center}
  \color{KCream}\itshape
  The recurring pattern: convert a messy position-by-time tensor into one scalar
  summary per gene, via linear algebra rather than hidden-state inference.
\end{center}
\end{frame}

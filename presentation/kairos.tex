% ============================================================ 02 KAIROS WORD ==
\begin{frame}{The word \smallKAIROS}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.58\textwidth}
  \vspace{0.4em}
  {\color{KGold}\bfseries\small Greek {\boldmath$\kappa\alpha\iota\rho\delta\varsigma$}.}\\[0.25em]
  {\color{KWhite}\small The opportune moment. The decisive instant when a
  change can be acted on --- distinct from \skyit{chronos}, the continuous
  flow of clock time.}
  \medskip

  {\color{KGold}\bfseries\small Why the name?}
  \begin{itemize}
    \setlength{\itemsep}{0.25em}
    \item Transcription is a chronological process --- polymerase grinds forward.
    \item But biological regulation happens at \creambf{kairotic} moments:
          a pause, a release, a hand-off between factors.
    \item The framework here is about \goldbf{catching those instants} in
          position along the gene body, not averaging them away.
  \end{itemize}
  \medskip

  {\color{KCream}\itshape\small
  Every base is a chance for the regulatory machinery to intervene.
  KAIROS is a language for those chances.}
\end{column}

  \centering

% ============================================================ 07 POL II KINETICS ==
\begin{frame}{RNA Polymerase II elongation kinetics}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.50\textwidth}
  \vspace{0.3em}
  {\color{KGold}\bfseries\small Mean velocity, and its limits.}\\[0.2em]
  {\color{KWhite}\small Classical single-molecule work (Singer, Darzacq, Core,
  Kwak) places \emph{average} Pol II elongation at roughly 1--4 kb/min.}
  \medskip

  {\color{KGold}\bfseries\small But the mean hides structure.}
  \begin{itemize}
    \setlength{\itemsep}{0.2em}
    \item Promoter-proximal pause: 20--60 bp downstream of TSS.
    \item Introns often faster than exons.
    \item Terminal slowdown near polyA sites (co-transcriptional cleavage).
    \item Pol II can backtrack --- kinetics have a diffusive component.
  \end{itemize}

% ============================================================ 09 WHY POSITION-RESOLVED ==
\begin{frame}{Why position-resolved speed?}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.52\textwidth}
  \vspace{0.3em}
  {\color{KWhite}\small
  RNA Polymerase II does not elongate at constant velocity. Over a single
  gene body it \creambf{pauses}, \creambf{accelerates}, and
  \creambf{decelerates} under competing molecular forces:}
  \begin{itemize}
    \setlength{\itemsep}{0.2em}
    \item nucleosome positioning and histone marks
    \item DNA sequence context and GC skew
    \item cotranscriptional splicing signals
    \item DNA damage and R-loop formation
    \item exon / intron boundaries
  \end{itemize}
  \vspace{0.6em}
  {\color{KCream}\itshape\small
  A single gene-wide rate erases the kinetic landscape that makes
  transcription biologically informative.}
\end{column}
\begin{column}{0.46\textwidth}
  \centering
  \vspace{0.6em}
  \begingroup\setbox0=\hbox{%

% ============================================================ 12 1/v PRINCIPLE ==
\begin{frame}{The $1/v$ principle}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.52\textwidth}
  \vspace{0.2em}
  {\color{KWhite}\footnotesize
  Pol II spends $1/v(x)$ units of time per unit length at $x$. Cumulative
  nascent-RNA at $x$ grows linearly in $t$, scaled by that dwell:}
  \vspace{-0.3em}
  \[
    \textstyle \mathbb{E}[\,r(x,t)\,]\;=\;\dfrac{c\,t}{v(x)}.
  \]
  \vspace{-0.7em}
  {\color{KWhite}\footnotesize So a straight-line fit in time at every position
  estimates $1/v$:}
  \vspace{-0.3em}
  \[
    \textstyle \widehat\beta(x) \;=\; \dfrac{c}{v(x)}
    \;\Longrightarrow\;
    \widehat v(x) \;\propto\; \dfrac{1}{\widehat\beta(x)}.
  \]
  \vspace{-0.6em}
  {\color{KCream}\itshape\scriptsize Slow regions accumulate reads quickly;
  fast regions stay sparse. The slope in time is the kinetic fingerprint.}
\end{column}

% ============================================================ 17 POSITIONAL REGRESSION ==
\begin{frame}{Positional regression: the estimator}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.52\textwidth}
  \vspace{0.1em}
  {\color{KWhite}\scriptsize For gene $g$, position bin $x$, and time points
  $t_i\in\{0,10,25,40\}$ min:}
  \vspace{-0.3em}
  \[
    \textstyle r_g(x,t_i) = \alpha_g(x) + \beta_g(x)\,t_i + \varepsilon_i .
  \]
  \vspace{-0.7em}
  {\color{KWhite}\scriptsize $\alpha_g(x)$ absorbs baseline depth; slope
  $\widehat\beta_g(x)$ is the \goldbf{positional accumulation rate} (PAR).
  By the $1/v$ principle:}
  \vspace{-0.3em}
  \[
    \widehat v_g(x) \;\propto\; 1/\widehat\beta_g(x).
  \]
  \vspace{-0.5em}
  {\scriptsize
  \begin{itemize}
    \setlength{\itemsep}{0.1em}
    \item Robust M-estimator (Huber loss)
    \item Smoothed with $\pm\,250$\,bp window
    \item Aggregated per gene via diversity $Z_M$
  \end{itemize}}

% ============================================================ 18 OLS VS HUBER ==
\begin{frame}{Robust regression matters}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.50\textwidth}
  \vspace{0.2em}
  {\color{KWhite}\footnotesize GRO-seq counts are heavy-tailed. A single PCR-duplicate
  spike at one time point can knock a slope off by 3$\sigma$.}

  {\color{KGold}\bfseries\footnotesize Huber loss:}
  \vspace{-0.3em}
  \[
    \textstyle L_c(u) = \begin{cases}
      \tfrac12 u^2, & |u|\le c \\
      c\,|u|-\tfrac12 c^2, & |u| > c
    \end{cases}
  \]
  \vspace{-0.6em}
  {\color{KMuted}\scriptsize tuning $c = 1.345\sigma$ gives 95\% efficiency at Gaussian noise.}

  {\color{KGold}\bfseries\footnotesize Why it helps here.}
  \begin{itemize}
    \setlength{\itemsep}{0.15em}
    \item Downweights the duplicate spike without zeroing it.
    \item Keeps the slope interpretable in read/minute units.
    \item No loss of power at clean positions.
  \end{itemize}
\end{column}

% ============================================================ 19 LOW-COUNT HANDLING ==
\begin{frame}{Low-count and zero-inflated positions}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.52\textwidth}
  \vspace{0.3em}
  {\color{KWhite}\small At early time points and distal positions, many bins
  register \skyit{zero reads}. A naive slope is meaningless there.}
  \medskip

  {\color{KGold}\bfseries\small Strategy.}
  \begin{itemize}
    \setlength{\itemsep}{0.2em}
    \item Classify each position as \skyit{pre-wave}, \skyit{transition}, or \skyit{elongated}
          using a three-state logistic on the cumulative counts.
    \item Fit $\widehat\beta(x)$ only on transition + elongated positions.
    \item Mark pre-wave positions as \creambf{missing} rather than zero.
  \end{itemize}
  \medskip

  {\color{KCream}\itshape\small This is the quiet step that keeps the
  spectral estimator honest: zeros before the wave front are not the same

% ============================================================ 21 GROUP THEORY MOTIVATION ==
\begin{frame}{A short detour: why algebra?}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.54\textwidth}
  \vspace{0.3em}
  {\color{KWhite}\small We need a \skyit{gene-level summary} of the vector
  $\widehat{\boldsymbol{\beta}}_g = (\widehat\beta(x_1),\ldots,\widehat\beta(x_M))$.}
  \medskip

  {\color{KGold}\bfseries\small A natural candidate --- the mean --- fails:}
  \begin{itemize}
    \setlength{\itemsep}{0.2em}
    \item Dominated by pause-spike outliers.
    \item Ignores concentration of signal on a few positions.
    \item Cannot distinguish one long slow patch from many brief ones.
  \end{itemize}
  \medskip

  {\color{KGold}\bfseries\small Algebraic diversity.}
  \begin{itemize}
    \setlength{\itemsep}{0.2em}
    \item View $\{\widehat\beta(x)\}$ as elements of an abelian group $(\mathbb{R},+)$.
    \item The diversity $Z_M$ counts distinct group elements, weighted by
          a continuous \skyit{resolution function}.
    \item The spectral $\psi$ captures the same idea via SVD.
  \end{itemize}
\end{column}
\begin{column}{0.42\textwidth}
  \vspace{0.3em}
  \kquote{Two genes can share a mean slope yet differ wildly in \emph{how} that
  mean is distributed across position. The diversity invariant is what tells
  them apart.}{Thornton \& Ray, 2022}
  \vspace{0.8em}

  {\color{KMuted}\scriptsize\itshape $Z_M$ was introduced in prior work on
  sequence-composition diversity; here it is transplanted to the
  position-resolved kinetic setting.}
\end{column}
\end{columns}
\end{frame}

% ============================================================ 22 ALGEBRAIC DIVERSITY ==
\begin{frame}{From position to gene: algebraic diversity}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.55\textwidth}
  \vspace{0.1em}
  {\color{KWhite}\scriptsize Per-position slopes are noisy.\ \ We want one
  scalar per gene that respects structure.}

  {\color{KGold}\bfseries\scriptsize Algebraic diversity $Z_M$.}
  \vspace{-0.2em}
  \[
    \textstyle Z_M = \dfrac{1}{M-1}\!\sum_{x=1}^{M}\!
       \mathbf{1}[\widehat\beta(x)\!\ne\!0]\,f(\widehat\beta(x)).
  \]
  \vspace{-0.6em}
  {\color{KMuted}\scriptsize $f$ a monotone resolution function; $Z_M$ counts
  effectively distinct slopes weighted by magnitude.}

  {\color{KGold}\bfseries\scriptsize Spectral concentration $\psi$.}
  \vspace{-0.2em}
  \[
    \textstyle \psi(g) = \dfrac{\lambda_1^{2}}{\sum_j \lambda_j^{2}}.
  \]
  \vspace{-0.7em}
  {\color{KMuted}\scriptsize $\{\lambda_j\}$: singular values of $R_g$.\ \
  Near $1$ $\Rightarrow$ one rank-one mode explains the signal: a clean wave.}
\end{column}

% ============================================================ 23 WHY SVD ==
\begin{frame}{Why SVD? The intuition behind $\psi$}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.50\textwidth}
  \vspace{0.1em}
  {\color{KWhite}\scriptsize The position-by-time read matrix $R_g\!\in\!\mathbb{R}^{M\times T}$
  factorises as a sum of rank-one terms:}
  \vspace{-0.3em}
  \[
    \textstyle R_g = \!\sum_{j=1}^{\min(M,T)}\! \lambda_j\,\mathbf{u}_j\,\mathbf{w}_j^{\!\top},
  \]
  \vspace{-0.6em}
  {\color{KMuted}\scriptsize $\lambda_j\!\ge\!0$ singular values; $\mathbf{u}_j$ position modes; $\mathbf{w}_j$ time modes.}

  {\color{KWhite}\scriptsize A coherent wave has \creambf{one} dominant mode.
  $\psi$ is the fraction of energy in that first mode:}
  \vspace{-0.3em}
  \[
    \psi(g) = \dfrac{\lambda_1^2}{\sum_j \lambda_j^2}\;\in\;(0,1].
  \]
  \vspace{-0.7em}
  {\color{KMuted}\scriptsize $\psi\!\to\!1$: one clean wave-front.\ \ $\psi\!\to\!0$: diffuse.}

  {\color{KCream}\itshape\scriptsize Dimensionless, scale-free, tracks wave-front speed.}
\end{column}

  corrupt the spectral summary.}

% ============================================================ 35 MIXED MODEL ==
\begin{frame}{Epigenetic covariates: the design}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.55\textwidth}
  \vspace{0.3em}
  {\color{KWhite}\small For each position $x$ in each gene $g$ we assemble a
  feature vector $\mathbf{z}_g(x)\in\mathbb{R}^{K}$:}
  \begin{itemize}
    \setlength{\itemsep}{0.2em}
    \item H3K4me3, H3K27ac, H3K36me3, H3K9me3, H3K27me3
    \item DNase-seq accessibility
    \item WGBS methylation ($\beta$-value)
    \item GC content, CpG density

% ============================================================ 45 APPENDIX FORMULAS ==
\begin{frame}{Appendix: formulas at a glance}
\vspace{0.6em}
\color{KWhite}\small
\begin{align*}
\text{Positional regression:}\quad &
   r_g(x,t) = \alpha_g(x) + \beta_g(x)\,t + \varepsilon,\qquad
   \widehat v_g(x) \propto 1/\widehat\beta_g(x) \\[0.3em]
\text{Algebraic diversity:}\quad &
   Z_M = \tfrac{1}{M-1}\sum_{x=1}^{M} \mathbf{1}[\widehat\beta(x)\ne 0]\,f(\widehat\beta(x)) \\[0.3em]
\text{Spectral concentration:}\quad &
   \psi(g) = \lambda_1^2 \big/ \textstyle\sum_j \lambda_j^2,\qquad
   \{\lambda_j\} = \text{SVD of}\ R_g \in \mathbb{R}^{M\times T} \\[0.3em]
\text{Epigenetic model:}\quad &
   \log \widehat v_g(x) = \mathbf{z}_g(x)^{\!\top}\boldsymbol\gamma + u_g + \eta(x) + \varepsilon \\[0.3em]
\text{Validation anchor:}\quad &
   v_g^{\text{DANKO}} = \text{wave-front location at time}\ t\ \big/\ t
\end{align*}
\vspace{0.8em}
\begin{center}
  {\color{KCream}\itshape The recurring pattern: convert a position-by-time tensor into one scalar summary per gene, via linear algebra rather than hidden-state inference.}
\end{center}
\end{frame}
